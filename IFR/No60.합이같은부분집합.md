#include <vector>
#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <utility>
using namespace std;

int sum = 0;
vector<int> nums;
int total=0;
bool flag=false;
int N;

void dfs(int x,int sum){
	if(sum>total/2) return;
	if(flag) return;
	if (x>N) {
		
		if(sum==(total-sum)){
			flag = true;
		}
		
	}else{
		dfs(x+1,sum+nums[x]);
		dfs(x+1,sum);
	}
	
}

int main() {
	
	int num;
	cin >> N;
	
	for(int i=0;i<N;i++){
		cin >> num;
		nums.push_back(num);
		total+=num;
	}
	
	dfs(1,0);
	
	if(flag) cout << "YES"<<endl;
	else cout << "NO"<<endl;


	return 0;
}
