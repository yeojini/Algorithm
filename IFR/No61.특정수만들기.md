```c++
#include <vector>
#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <utility>
using namespace std;

vector<int> nums;
int N,M;
int cnt = 0;

void dfs(int x,int sum){
	if(x==N) {
		if(sum == M) cnt++;
		return;
	}
	else{
		dfs(x+1,sum+nums[x]);
		dfs(x+1,sum);
		dfs(x+1,sum-nums[x]);
	}
	
}
int main() {

	int num;
	cin >> N >> M;
	for(int i=0;i<N;i++){
		cin >> num;
		nums.push_back(num);
	}
	
	dfs(0,0);
	
	if(cnt==0) cout << -1;
	else cout << cnt;


	return 0;
}
```
