# 문제
프로그래머스 코딩테스트연습 스택/큐 기능개발
C/C++ 문제 풀이

# 풀이

# 코드
```cpp
#include <string>
#include <vector>
#include <queue>
using namespace std;

vector<int> solution(vector<int> progresses, vector<int> speeds) {
    vector<int> answer;
    int idx = 0;
    int time = 0;
    queue<int> q;
    int cnt = 0;
    
    for(int i=0;i<progresses.size();i++){
        q.push(i);
    }
    
    while(!q.empty()){
        
        for(int i=0;i<progresses.size();i++){
            progresses[i]+=speeds[i];
        }
        
        for(int i=0;i<progresses.size();i++){
           if(i==q.front() && progresses[i]>=100){
                q.pop();
                cnt++;
           }
        }
        
        if(cnt!=0) answer.push_back(cnt);
        cnt = 0;
        
    }
        
      
    
    
    return answer;
}
```
