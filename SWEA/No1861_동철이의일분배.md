# 1. 풀이

- 소수점 6자리 출력
```
    cout.setf(ios::fixed);
    cout.precision(6);
```
- 2차원 배열에 입력 받으면서 확률 (/100) 로 변환
- dfs 의 인자로 확률을 전달
- 분배가 된 일은 다시 분배되면 안되므로 visited 배열

# 2. 코드
```
    #include <iostream>
    
    using namespace std;
    
    int T, N;
    double maxx = 0;
    double arr[17][17];
    bool visited[17];
    
    void solve(int depth, double prob) {
    	if (depth == N) {
    		if (maxx < prob) {
    			maxx = prob;
    		}
    		return;
    	}
    
    	if (prob <= maxx) return;
    
    	for (int i = 0; i < N; i++) {
    		if (!visited[i] && arr[depth][i] != 0) {
    			visited[i] = true;
    			solve(depth + 1, prob*arr[depth][i]);
    			visited[i] = false;
    		}
    	}
    
    
    }
    
    int main() {
    
    	ios::sync_with_stdio(0);
    	cin.tie(0);
    	cout.tie(0);
    	cout.setf(ios::fixed);
    	cout.precision(6);
    	//소수점 6자리 고정
    
    	cin >> T;
    
    	for (int tc = 1; tc <= T; tc++) {
    
    		cin >> N;
    
    		for (int i = 0; i < 17; i++) {
    			for (int j = 0; j < 17; j++) {
    				arr[i][j] = 0;
    			}
    			visited[i] = false;
    		}
    		maxx = 0;
    
    		for (int i = 0; i < N; i++) {
    			for (int j = 0; j < N; j++) {
    				cin >> arr[i][j];
    				arr[i][j] /= 100;
    			}
    		}
    
    		solve(0, 1);
    
    		cout << "#" << tc << " " << maxx * 100 << "\n";
    
    	}
    	return 0;
    }
```
