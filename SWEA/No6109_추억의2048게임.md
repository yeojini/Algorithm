## 1. 풀이

- 다음 좌표가 밀었던 좌표이면 덧셈 수행 안한다 <- check[20][20] 선언
- 입력받은 문자열 S 에 따라 map  내의 탐색 좌표의 방향, 다음 좌표가 변한다
- 재귀 함수

## 2. 코드
```
#include <iostream>
#include <string>

using namespace std;

int N;
int** map;
bool check[20][20] = { false, };
int dx[] = { -1,0,1,0 };
int dy[] = { 0,1,0,-1 }; //위,오른쪽,아래,왼쪽

void move(int row, int col, int dir) {
	int nx = row + dx[dir];
	int ny = col + dy[dir];
	if (nx >= 0 && ny >= 0 && nx < N && ny < N) { //map 범위 적용
		if (map[nx][ny] == 0) {
			map[nx][ny] = map[row][col];
			map[row][col] = 0; //다음 자리가 0이면 옮기고 원래 자리를 0 으로 만든다
			move(nx, ny, dir);
		}
		else if (map[nx][ny] > 0 && !check[nx][ny]) { //다음 자리가 더해졌던 애가 아닐때만 더하는 과정 수행
			check[nx][ny] = true;
			if (map[nx][ny] == map[row][col]) { //다음 자리와 현재 자리 수 가 같으면
				map[nx][ny] += map[row][col]; //다음 자리에 현재 자리수와 같은 값을 저장하고
				map[row][col] = 0; //현재 자리수를 0으로 만든다.
				move(nx, ny, dir);
			}
		}
	}
}

void solve(string S) {

	if (S == "up") {
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (map[i][j] != 0)
					move(i, j, 0);
			}
		}
	}
	else if (S == "right") {
		for (int i = N - 1; i >= 0; i--) {
			for (int j = 0; j < N; j++) {
				if (map[j][i] != 0)
					move(j, i, 1);
			}
		}
	}
	else if (S == "down") {
		for (int i = N - 1; i >= 0; i--) {
			for (int j = 0; j < N; j++) {
				if (map[i][j] != 0)
					move(i, j, 2);
			}
		}
	}
	else if (S == "left") {
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (map[j][i] != 0)
					move(j, i, 3);
			}
		}
	}
}


int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	int T;
	string S;
	cin >> T;
	for (int tc = 1; tc <= T; tc++) {
		cin >> N >> S;

		map = new int*[N];
		for (int i = 0; i < N; i++)
			map[i] = new int[N];
		//이차원 배열의 동적 할당 

		for (int i = 0; i < 20; i++) {
			for (int j = 0; j < 20; j++) check[i][j] = false;
		}

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				int tmp;
				cin >> tmp;
				map[i][j] = tmp;
			}
		}
		solve(S);

		cout << "#" << tc << "\n";



		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				cout << map[i][j] << " ";
			}
			cout << "\n";
		}
	}
	return 0;
}
```
